\section{Multi-Agent Systems}

\begin{frame}{Definition}
\begin{itemize}
  \item many interacting autonomous entities
  \item each entity only uses subset of relevant information
  \item humble computational and/or physical capabilities
  \item key: interaction between agents
  \item \cite{panait2005cooperative,ferber2003agents}
\end{itemize}
\end{frame}

\begin{frame}{Applications}
\begin{itemize}
  \item modeling natural systems
  \item engineering distributed systems
  \item algorithm design
\end{itemize}
\end{frame}
